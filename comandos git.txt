comandos git:

1 - git clone <url>
	copia o repositório da nuvem

2- git add .
	adiciona as mudanças na linha do tempo

3 - git commit -m "<mudanças feitas>"
	cria um nome pra mudanças

4- git push
	envia mudanças pra nuvem

5- git pull 
	baixa mudanças da nuvem 

6- git status
	mostra mudanças locais

7- git fetch
	baixa as mudanças da nuvem
